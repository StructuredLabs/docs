---
title: "Delete Block"
api: "POST https://api.structuredlabs.io/delete-block"

<ParamField query="apiToken" type="string" optional="false">
  The API token for authentication.
</ParamField>

<ParamField query="blockId" type="string" optional="false">
  The identifier for the block.
</ParamField>

### Response

<ResponseField name="success" type="boolean">
  Indicates whether the call was successful. Returns `true` if successful,
  `false` if not.
</ResponseField>

